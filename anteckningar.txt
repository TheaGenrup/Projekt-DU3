    console.log(activeTab.toLowerCase());

        console.log(tab.textContent.toLowerCase().replace(" ", ""));
        if (tab.textContent.toLowerCase().replace(" ", "") == activeTab.toLowerCase()) {
            tab.classList.add("active")

            tab.classList.add("inactive")
        }
    });

}
